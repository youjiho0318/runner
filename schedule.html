function doGet(e) { var TZ = "Asia/Seoul"; var action = (e && e.parameter && e.parameter["action"]) || "schedule"; if (action === "getHidden") return getHiddenDrivers(); if (action === "setHidden") { var data = (e && e.parameter && e.parameter["data"]) || "[]"; return setHiddenDrivers(data); } // ── 캠프위치 실시간 조회 (캐시 없음) ── if (action === "getCampLocations") return getCampLocations(); // ── 오늘(PDD) 기준 이번주 일요일 ── var now = new Date(); now.setDate(now.getDate() + 1); var today = Utilities.formatDate(now, TZ, "yyyy-MM-dd"); var todayDate = new Date(today + "T00:00:00"); var thisSun = new Date(todayDate); thisSun.setDate(todayDate.getDate() - todayDate.getDay()); // 범위: 2주 전 일요일 ~ 다음주 토요일 (28일 고정) var rangeStart = new Date(thisSun); rangeStart.setDate(thisSun.getDate() - 14); var rangeEnd = new Date(thisSun); rangeEnd.setDate(thisSun.getDate() + 13); var startDs = Utilities.formatDate(rangeStart, TZ, "yyyy-MM-dd"); var endDs = Utilities.formatDate(rangeEnd, TZ, "yyyy-MM-dd"); // ── 캐시 (6분) ── var cacheKey = "sch_" + startDs + "_" + endDs; var cache = CacheService.getScriptCache(); var cached = cache.get(cacheKey); if (cached) { return ContentService.createTextOutput(cached).setMimeType(ContentService.MimeType.JSON); } // ── 날짜 목록 (28일) ── var targetDates = [], targetSet = {}; var cur = new Date(rangeStart); while (Utilities.formatDate(cur, TZ, "yyyy-MM-dd") <= endDs) { var ds = Utilities.formatDate(cur, TZ, "yyyy-MM-dd"); targetDates.push(ds); targetSet[ds] = true; cur.setDate(cur.getDate() + 1); } var sApp = SpreadsheetApp.getActiveSpreadsheet(); // ── 공지 ── var msg = ""; try { var nsh = sApp.getSheetByName("공지"); if (nsh) msg = String(nsh.getRange(1,1).getValue()||"").trim(); } catch(ex) {} // ── 백업 시트 ── var bkSheet = sApp.getSheetByName("백업"); if (!bkSheet || bkSheet.getLastRow() < 3) { var empty = JSON.stringify({drivers:[], notice:msg, dates:targetDates}); cache.put(cacheKey, empty, 360); return ContentService.createTextOutput(empty).setMimeType(ContentService.MimeType.JSON); } var bkLastCol = bkSheet.getLastColumn(); var bkLastRow = bkSheet.getLastRow(); // ── 1행 헤더에서 필요 컬럼만 추출 (전체 읽기 X) ── var hdr = bkSheet.getRange(1, 1, 1, bkLastCol).getValues()[0]; var needCols = [0], needDates = [null]; for (var ci = 1; ci < hdr.length; ci++) { var m = String(hdr[ci]).match(/(\d{4}-\d{2}-\d{2})/); if (m && targetSet[m[1]]) { needCols.push(ci); needDates.push(m[1]); } } // ── 스케쥴 시트에서 camp/id/route ── var infoMap = {}; try { var sch = sApp.getSheetByName("스케쥴") || sApp.getSheets()[0]; var sLast = sch.getLastRow(); if (sLast >= 10) { var sData = sch.getRange(10, 3, sLast - 9, 4).getValues(); for (var si = 0; si < sData.length; si++) { var nm = String(sData[si][1]||"").trim(); if (nm) infoMap[nm] = { camp: String(sData[si][0]||"").trim(), id: String(sData[si][2]||"").trim(), route: String(sData[si][3]||"").trim() }; } } } catch(ex) {} // ── 필요 컬럼 범위만 1회 읽기 ── var minC = Math.min.apply(null, needCols); var maxC = Math.max.apply(null, needCols); var dataRows = bkSheet.getRange(3, minC + 1, bkLastRow - 2, maxC - minC + 1).getValues(); var relIdx = needCols.map(function(c){ return c - minC; }); var drivers = []; for (var ri = 0; ri < dataRows.length; ri++) { var row = dataRows[ri]; var nm = String(row[relIdx[0]]||"").trim(); if (!nm || nm === "날짜" || nm.indexOf("최근") > -1) continue; var sched = {}; for (var di = 1; di < relIdx.length; di++) { sched[needDates[di]] = String(row[relIdx[di]]||"").trim(); } var info = infoMap[nm] || {camp:"", id:"", route:""}; drivers.push({camp:info.camp, name:nm, id:info.id, route:info.route, schedule:sched}); } var result = JSON.stringify({drivers:drivers, notice:msg, dates:targetDates}); try { cache.put(cacheKey, result, 360); } catch(ex) {} return ContentService.createTextOutput(result).setMimeType(ContentService.MimeType.JSON); } // ── 캠프위치 실시간 조회 (캐시 없음 — 시트 수정 즉시 반영) ── function getCampLocations() { try { var sApp = SpreadsheetApp.getActiveSpreadsheet(); var sh = sApp.getSheetByName("캠프위치"); if (!sh || sh.getLastRow() < 2) { return ContentService.createTextOutput(JSON.stringify({ok:true, locations:{}})) .setMimeType(ContentService.MimeType.JSON); } var rows = sh.getRange(2, 1, sh.getLastRow() - 1, 2).getValues(); var locations = {}; for (var i = 0; i < rows.length; i++) { var campName = String(rows[i][0]||"").trim(); var address = String(rows[i][1]||"").trim(); if (campName && address) locations[campName] = address; } return ContentService.createTextOutput(JSON.stringify({ok:true, locations:locations})) .setMimeType(ContentService.MimeType.JSON); } catch(ex) { return ContentService.createTextOutput(JSON.stringify({ok:false, locations:{}, error:ex.message})) .setMimeType(ContentService.MimeType.JSON); } } function getHiddenDrivers() { try { var cache = CacheService.getScriptCache(); var cached = cache.get("hidden_drivers"); if (cached) return ContentService.createTextOutput(JSON.stringify({ok:true, names:JSON.parse(cached)})).setMimeType(ContentService.MimeType.JSON); var sApp = SpreadsheetApp.getActiveSpreadsheet(); var sh = sApp.getSheetByName("숨김설정"); if (!sh || sh.getLastRow() < 1) return ContentService.createTextOutput(JSON.stringify({ok:true, names:[]})).setMimeType(ContentService.MimeType.JSON); var vals = sh.getRange(1, 1, sh.getLastRow(), 1).getValues(); var names = vals.map(function(r){ return String(r[0]||"").trim(); }).filter(Boolean); cache.put("hidden_drivers", JSON.stringify(names), 3600); return ContentService.createTextOutput(JSON.stringify({ok:true, names:names})).setMimeType(ContentService.MimeType.JSON); } catch(ex) { return ContentService.createTextOutput(JSON.stringify({ok:false, names:[]})).setMimeType(ContentService.MimeType.JSON); } } function setHiddenDrivers(dataJson) { try { var sApp = SpreadsheetApp.getActiveSpreadsheet(); var sh = sApp.getSheetByName("숨김설정"); if (!sh) sh = sApp.insertSheet("숨김설정"); sh.clearContents(); var names = JSON.parse(dataJson); if (names.length > 0) sh.getRange(1, 1, names.length, 1).setValues(names.map(function(n){ return [n]; })); var cache = CacheService.getScriptCache(); cache.put("hidden_drivers", dataJson, 3600); return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON); } catch(ex) { return ContentService.createTextOutput(JSON.stringify({ok:false, error:ex.message})).setMimeType(ContentService.MimeType.JSON); } }
